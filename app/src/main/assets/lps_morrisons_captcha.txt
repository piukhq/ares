var recaptchaAnchorInvalidQuery = "re-captcha[class*=-invalid]"

performLogin()

function performLogin() {
    var rInvalid = document.querySelector(recaptchaAnchorInvalidQuery)

    if (rInvalid) {
        return {
            "success": false
        }
    }

    return {
        "success": true
    }
}
